def mod_string(col):
    """modifies a given string"""
    if 'Day' in col:  # ScheduledDay, AppointmentDay
        col = (col[:-3] + '_' + col[-3:])  # split the name and put _ 
    if 'No-show' in col :
        col = 'No_show'
    return col
df.rename(columns = lambda x: mod_string(x),inplace=True)
